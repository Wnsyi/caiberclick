# 使用轻量级的 Node.js 环境
FROM node:22-alpine

# 安装 electron-builder 在 Linux 上打包 Windows 应用的必备工具
RUN apk add --no-cache wine

# 设置工作目录
WORKDIR /project

# 复制项目配置文件并安装依赖
COPY package.json ./
RUN npm install

# 设置 Electron 镜像加速
ENV ELECTRON_MIRROR=https://npmmirror.com/mirrors/electron/
ENV ELECTRON_CUSTOM_DIR=v41.5.0

# 复制整个 electron 目录下的源码和资源
COPY . .

# 执行打包命令
CMD ["npm", "run", "build"]
